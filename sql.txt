-- Questão 1
SELECT c.Nome,
       SUM(p.ValorTotalpedido) AS TotalGasto
FROM cliente c
INNER JOIN pedido p
    ON c.idcliente = p.cliente_idcliente
GROUP BY c.idcliente, c.Nome
ORDER BY TotalGasto DESC
LIMIT 1;

-- Questão 2
SELECT c.Descricao AS Categoria,
       p.Nome AS Produto,
       SUM(php.Quantidade) AS TotalVendido
FROM categoria c
INNER JOIN produto p
    ON c.idcategoria = p.categoria_idcategoria
INNER JOIN pedido_has_produto php
    ON p.idproduto = php.produto_idproduto
GROUP BY c.Descricao, p.Nome
HAVING SUM(php.Quantidade) = (
    SELECT MAX(TotalQtd)
    FROM (
        SELECT SUM(php2.Quantidade) AS TotalQtd
        FROM produto p2
        INNER JOIN pedido_has_produto php2
            ON p2.idproduto = php2.produto_idproduto
        WHERE p2.categoria_idcategoria = c.idcategoria
        GROUP BY p2.idproduto
    ) AS sub
);

-- Questão 3
SELECT c.Nome,
       COUNT(p.idpedido) AS QuantidadePedidos,
       SUM(p.ValorTotalpedido) AS TotalPedidos
FROM cliente c
INNER JOIN pedido p
    ON c.idcliente = p.cliente_idcliente
GROUP BY c.idcliente, c.Nome
HAVING COUNT(p.idpedido) > 2
   AND SUM(p.ValorTotalpedido) > 5000;

-- Questão 4
SELECT c.Descricao AS Categoria,
       SUM(php.Quantidade * php.PrecoUnitario) AS TotalVendas
FROM categoria c
INNER JOIN produto p
    ON c.idcategoria = p.categoria_idcategoria
INNER JOIN pedido_has_produto php
    ON p.idproduto = php.produto_idproduto
GROUP BY c.idcategoria, c.Descricao
ORDER BY TotalVendas DESC
LIMIT 1;

-- Questão 5
SELECT t.Descricao AS TipoCliente,
       AVG(PedidosPorCliente.TotalPedidos) AS MediaPedidos
FROM (
    SELECT c.idcliente,
           c.tipocliente_idtipocliente,
           COUNT(p.idpedido) AS TotalPedidos
    FROM cliente c
    LEFT JOIN pedido p
        ON c.idcliente = p.cliente_idcliente
    GROUP BY c.idcliente, c.tipocliente_idtipocliente
) AS PedidosPorCliente
INNER JOIN tipocliente t
    ON PedidosPorCliente.tipocliente_idtipocliente = t.idtipocliente
GROUP BY t.idtipocliente, t.Descricao;
